# PUT requests
curl -X PUT --data-binary "@test_images/cat.200.jpg" http://localhost:3000/200
curl -X PUT --data-binary "@test_images/cat.404.jpg" http://localhost:3000/404
curl -X PUT --data-binary "@test_images/cat.500.jpg" http://localhost:3000/500

# GET requests
curl -X GET http://localhost:3000/200 --output output200.jpg
curl -X GET http://localhost:3000/404 --output output404.jpg
curl -X GET http://localhost:3000/500 --output output500.jpg
